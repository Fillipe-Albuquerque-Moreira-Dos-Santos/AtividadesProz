def calcular_idade(ano_nascimento):
    from datetime import date
    return date.today().year - ano_nascimento

def obter_ano_valido():
    while True:
        ano_str = input("Digite o ano de nascimento (entre 1922 e 2021): ")
        if ano_str.isdigit() and 1922 <= int(ano_str) <= 2021:
            return int(ano_str)
        print("Ano inválido. Tente novamente.")

print("Programa de Cálculo de Idade\n")
nome_completo = input("Digite o seu nome completo: ")

ano_nascimento = obter_ano_valido()
idade = calcular_idade(ano_nascimento)

print(f"\nOlá, {nome_completo}! Você completou ou irá completará {idade} anos em 2022.")
